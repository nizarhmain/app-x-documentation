export class Cliente{
    public id: number;
    public codice: string;
    public ragionesociale: string;
    public indirizzo: string;
    public comune: string;
    public telefono: string;
    public email: string;
    public banca: string;
    public pagamento: string;
    public sconto1: number;
    public sconto2: number;
    public sconto3: number;
    public listino: string;
    public saldo: number;
    public lastupdate: string;
    public eliminato: number;
    public comp_code: string;
    public codice_agente: string;

    public constructor(codice: string, ragionesociale: string, indirizzo: string, comune: string, telefono: string, email: string,
                       banca: string, pagamento: string, sconto1: any, sconto2: any, sconto3: any, listino: string, saldo: number,
                       id: number = 0, lastupdate: string = '', eliminato: any = 0, comp_code: string = '', codice_agente: string = '') {
        this.id = id ? id: 0;
        this.codice = codice;
        this.ragionesociale = ragionesociale;
        this.indirizzo = indirizzo;
        this.comune = comune;
        this.telefono = telefono;
        this.email = email;
        this.banca = banca;
        this.pagamento = pagamento;
        this.sconto1 = this.parseSconto(sconto1);
        this.sconto2 = this.parseSconto(sconto2);
        this.sconto3 = this.parseSconto(sconto3);
        this.listino = listino;
        this.saldo = saldo;
        this.lastupdate = lastupdate ? lastupdate : '';
        this.eliminato = eliminato ? 1 : 0;
        this.comp_code = comp_code ? comp_code : '';
        this.codice_agente = codice_agente ? codice_agente : '';
    }

    private parseSconto(sconto: any): number {
        if(!sconto) {
            return 0;
        }
        else {
            if((typeof sconto) == "string") {
                return parseFloat(sconto.replace(",", "."));
            }
            else {
                if((typeof sconto) == "number") {
                    return sconto;
                }
                else {
                    return 0;
                }
            }
        }
    }

    getSconto1() {
        return this.sconto1.toFixed(2);
    }

    getSconto2() {
        return this.sconto2.toFixed(2);
    }

    getSconto3() {
        return this.sconto3.toFixed(2);
    }
}
