import { StoreClientsInRamService } from 'src/app/services/store-clients-in-ram.service';

import { CallNumber } from '@ionic-native/call-number/ngx';

@Component({
  selector: 'app-cliente-info',
  templateUrl: './cliente-info.page.html',
  styleUrls: ['./cliente-info.page.scss'],
})
export class ClienteInfoPage implements OnInit {

  loadedCliente: Cliente;
  
  constructor(
    private db: DbProvider,
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private storeInRam: StoreClientsInRamService,
    private callNumber: CallNumber
  ) { } 

  ngOnInit() {
    console.log('cliente info ngOnInit');
    this.activatedRoute.paramMap.subscribe(paramMap => {
      if (!paramMap.has('codice')) {
        // redirect
        console.log('shits on fire yo');
        return;
      }
      let codice = paramMap.get('codice');

      this.loadedCliente = this.storeInRam.getCliente(codice)
    });
  }

  chiamaCliente() {
    this.callNumber.callNumber(this.loadedCliente.telefono, true)
  .then(res => console.log('Launched dialer!', res))
  .catch(err => console.log('Error launching dialer', err));
  }


}
