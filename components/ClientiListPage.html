import { Component, OnInit } from "@angular/core";
import { ToolsProvider } from "src/app/services/tool-provider";
import { DbProvider } from "src/app/services/db-provider";
import { Cliente } from "src/app/app.model";
import { Platform, NavController } from "@ionic/angular";
import { Router, NavigationExtras } from '@angular/router';
import { StoreClientsInRamService } from 'src/app/services/store-clients-in-ram.service';

@Component({
  selector: "app-clienti-list",
  templateUrl: "./clienti-list.page.html",
  styleUrls: ["./clienti-list.page.scss"]
})
export class ClientiListPage implements OnInit {
  mode: number;
  pageTitle: string;
  items: Array<Cliente>;
  listOrder: string;
  listino: string = "";
  textFilter: string = "";

  constructor(
    public tools: ToolsProvider,
    public db: DbProvider,
    public plt: Platform,
    public router: Router,
    public navCtrl: NavController,
    public storeInRam: StoreClientsInRamService
  ) {
  }

  ionViewWillEnter() {
    console.log('loading please'); 
    console.log(this.items); 
  }


  ngOnInit() {

    if (this.storeInRam.areClientsInMemory() == true) {
      this.items = this.storeInRam.getAllClienti()      
    } else {
      this.initializeItems(); 
    }

  }


  async initializeItems() {
    //let loader = this.tools.getLoader('Caricamento Clienti...');
    this.db.getAllClienti().then(clienti => {
      this.items = clienti;

      this.storeInRam.setClienti(clienti);

      //console.log(clienti);
    });
  }
}
