import { Component } from '@angular/core';
import { AlertController, MenuController, NavController, LoadingController } from '@ionic/angular';

import { GlobalProvider } from '../../services/global-provider';
import { DbProvider } from '../../services/db-provider';
import { FileProvider } from '../../services/file-provider';
import { ToolsProvider } from '../../services/tool-provider';

import { Settings } from '../../app.model';

//import { LogViewerPage } from '../log-viewer/log-viewer';

  selector: 'page-settings',

  /*DEBUGSTRING*/

  pageTitle: string;
  settings: Settings;
  version: string;
  dbversion: string;
  build: string;

  constructor(public navCtrl: NavController,
              public alertCtrl: AlertController,
              public menuCtrl: MenuController,
              public db: DbProvider,
              public file: FileProvider,
              public loadCtrl: LoadingController,
              public tools: ToolsProvider,
              public global: GlobalProvider) {
    global.log('SettingsPage.contructor() - started');
    this.pageTitle = 'Impostazioni';
    //this.initialize();

  }


  ngOnInit(){

    console.log('ng on init');

    // if the settings are not existing set them to empty otherwise angular throws an error where it tries to bind an undefined value to a ngModel bind
    if(this.global.getSettings() === undefined) {
      // initaliaze the settings to empty
      this.settings = new Settings("", "", "", "","","","","");
      this.global.setSettings(this.settings);
    }

    // otherwise continue normally
    // fetch the settings saved in memory
    this.settings = this.global.getSettings();
    this.version = this.global.getAppVersion();
    this.db.dbVer().then(dbver => this.dbversion = dbver);
    this.build = this.global.getAppBuild();

    //this.settings = new Settings("", "", "campomaggio", "9n3curu6","https://www.tecnoin.eu/TecnoIn/REST2","","","");


  }

  success() {
    console.log(' i got clicked');
  }


  async saveTapped(ev) {

    const loading = await this.loadCtrl.create({
			message: 'Saving settings',
		  });

		await loading.present();

    // save the settings to SQLITE
    this.db.saveSettings(this.settings).then(async() => {
      this.global.setSettings(this.settings);
      await loading.dismiss();
    });

    this.global.log('SettingsPage.saveTapped() - started');
  }
