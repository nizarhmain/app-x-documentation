import { Component, OnInit } from "@angular/core";
import {
  NavController,
  LoadingController,
  MenuController,
  AlertController,
  Platform,
  Events
} from "@ionic/angular";
import { DbProvider } from "src/app/services/db-provider";
import { ActivatedRoute, Router} from '@angular/router';
import { ToolsProvider } from "src/app/services/tool-provider";
import { GlobalProvider } from "src/app/services/global-provider";
import { FileProvider } from "src/app/services/file-provider";
import { SyncProvider } from "src/app/services/sync-provider";
import { RestProvider } from "src/app/services/rest-provider";

@Component({
  selector: "app-home",
  templateUrl: "./home.page.html",
  styleUrls: ["./home.page.scss"]
})
export class HomePage implements OnInit {
  ngOnInit(): void {
    console.log('home, this must get called only one time');

  }
  constructor(
    private route: ActivatedRoute,
    public navCtrl: NavController,
    public loadCtrl: LoadingController,
    public menuCtrl: MenuController,
    public alertCtrl: AlertController,
    public platform: Platform,
    public db: DbProvider,
    public tools: ToolsProvider,
    public global: GlobalProvider,
    public file: FileProvider,
    public sync: SyncProvider,
    public events: Events,
    public rest: RestProvider
  ) {

    console.log('home, this must get called only one time');

    this.backButtonEvent();


    platform.ready().then(readySource => {
      global.log('HomePage.constructor() - PlatformReady: ' + readySource);
      db.ready().then(() => this.initialize());
    });

  }

  
  backButtonEvent() {
    
    this.platform.backButton.subscribe( () => {
      console.log('hardware button touched');
      this.navCtrl.back()
    })
    
  }

  initialize() {
    console.log('this is not loaded from the first time, we probably need to add it as an entry component');
    this.db.loadSettings().then(settings => {
      this.global.setSettings(settings);
    });
  }
}
