  initialize() {
    // need to get fixed
    this.db.getNumClienti().then(totClienti => {
      this.totClienti = totClienti;
      this.db.getNumDestinatari().then(totDestinatari => {
        this.totDestinatari = totDestinatari;
        this.db.getNumArticoli().then(totArticoli => {
          this.totArticoli = totArticoli;
          this.db.getNumPreferiti().then(totPreferiti => {
            this.totPreferiti = totPreferiti;
            this.db.getNumImmagini().then(totMetaImmagini => {
              this.totMetaImmagini = totMetaImmagini;
              this.db.getNumOrdini().then(totOrdini => {
                this.totOrdini = totOrdini;
                this.db.getNumOrdiniDaInviare().then(totOrdiniDaInviare => {

   async syncDatiTap() {
    let alertStart = await this.alertCtrl.create({
      header: 'Sync Database',
      message: 'Avviare la sincronizzazione dei dati?',
      mode: 'ios',
      cssClass: 'syncAlert',
      buttons: [
        {
          text: 'No',
          cssClass: 'syncNoBtn',
          role: 'cancel',
          handler: () => {
            this.global.log('SyncPage.syncDatiTap() - StartAlert: No clicked');
          }
        },
        {
          text: 'Si',
          cssClass: 'syncYesBtn',
          handler: async () => {

            this.global.log('SyncPage.syncDatiTap() - StartAlert: Si clicked');

            this.sync.updateDBdata().then(async() => {

              this.global.log('SyncPage.syncDatiTap() - Dati aggiornati');
              let alertOK = await this.alertCtrl.create({
                header: 'Aggiornamento',
                cssClass: 'popOk',
                mode: 'ios',
                message: 'Aggiornamento dati completato',
                buttons: [
                  {
                    text: 'OK',
                    role: 'cancel',
                    handler: () => { this.initialize(); }
                  }
                ]
              });
              alertOK.present();
            }).catch(async(err) => {
              this.global.log('SyncPage.syncDatiTap() - Error: ' + err.message);
              let alertErr = await this.alertCtrl.create({
                header: 'Errore durante l\'aggiornamento',
                cssClass: 'popOk',
                mode: 'ios',
                message: err.message,
                buttons: [
                  {
                    text: 'OK',
                    role: 'cancel'
                  }
                ]
              });
              alertErr.present();
            });
          }
        }
      ]
    });
    alertStart.present();
  }
