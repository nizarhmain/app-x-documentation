  constructor(public platform: Platform,
              public file: File,
              public fileTransfer: FileTransfer,
              public global: GlobalProvider,
              public tools: ToolsProvider,
              public db: DbProvider) {
    global.log('FileProvider.constructor() - started');
    platform.ready().then((readySource) => {
      global.log('FileProvider.constructor() - PlatformReady: ' + readySource);
      this.initialize();
    });
  }

  initialize() {
    this.global.log('FileProvider.initialize() - started');
  }

  getImageBase64(imgName: string): Promise<string> {
    let dataDir: string = cordova.file.dataDirectory;
    let imgDir: string = dataDir + 'Images';
    this.global.log('FileProvider.getImageBase64() - imgDir: ' + imgDir + ', imgName: ' + imgName);
    return this.file.readAsDataURL(imgDir, imgName);
  }

  getImagesDir(): Promise<string> {
    return this.getDir('Images');
  }

  getDir(dir: string): Promise<string> {
    let dataDir: string = cordova.file.dataDirectory;
    this.global.log("FileProvider.getDir('" + dir + "') - started");
    return this.file.createDir(dataDir, dir, false).then((dirEntry) => {
      this.global.log("FileProvider.getDir('" + dir + "') - Directory creata");
      return Promise.resolve(dataDir + dir + '/');
    }).catch((err) => {
      this.global.error("FileProvider.getDir('" + dir + "') - CreateDirError: " + err.message);
      if(err.message == 'PATH_EXISTS_ERR') {
        return Promise.resolve(dataDir + dir + '/');
      }
      else {
        let myerr = this.tools.parseError(err, 1, 'Errore Directory');
        this.global.log('FileProvider.getDir() - Error: ' + myerr.message);
        return Promise.reject(myerr);
      }
    });
  }

  getDataDir(): string {
    return cordova.file.dataDirectory;
  }

  
  listDir(dir: string): any {
    let dataDir: string = cordova.file.dataDirectory;
    this.global.log('FileProvider.listDir() - started');
    return this.file.listDir(dataDir, dir);
  }
  

  removeFile(dir, name): Promise<any> {
    let dataDir: string = cordova.file.dataDirectory;
    this.global.log('FileProvider.removeFile() - started');
    return this.file.removeFile(dataDir, dir + name);
  }
