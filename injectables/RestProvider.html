  logout(): Promise<any> {
    this.global.log('RestProvider.logout() - started');
    let url = this.global.getSettings().server + '/api.php';
    return new Promise((resolve, reject) => {
      this.http.post(url, {}).toPromise().then(resp => {
        this.global.log('RestProvider.logout() - logged out');
        resolve();
      }).catch(err => {
        this.global.log('RestProvider.logout() - logged out');
        reject(err);
      });
    });
  }

  loadTestate(testate: any[]): Promise<any> {
    this.global.log('RestProvider.loadTestate() - started');
    return new Promise((resolve, reject) => {
      this.login().then(csrf => {
        this.global.log('RestProvider.loadTestate() - Logged in, csrf: ' + csrf);
        let url = this.global.getSettings().server + '/api.php/testata?csrf=' + csrf;
        this.http.post(url, JSON.stringify(testate)).toPromise().then(resp => {
          this.global.log('RestProvider.loadTestate() - Caricamento Testate effettuato');
          this.logout().then(() => {
            resolve();
          });
        });
      }).catch(err => {
        this.global.log('RestProvider.loadTestate() - error: ' + err);
        reject(err);
      });
    });
  }
