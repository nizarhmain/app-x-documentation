  /** After check the function returns:
   *  - cancel: if the user press the cancel button
   *  - ok: if the user insert the correct key
   *  - ko: if the user insert a wrong key
   */
  passwordChecker(): Promise<string> {
    this.global.log('ToolsProvider.passwordChecker() - started');
    let uuid: string = this.device.uuid;
    let d: Date = new Date();
    let year: string = d.getFullYear().toString();
    let month: string = d.getMonth() + 1 > 9 ? (d.getMonth() + 1).toString() : '0' + (d.getMonth() + 1).toString();
    let day: string = d.getDate() > 9 ? d.getDate().toString() : '0' + d.getDate().toString();
    let hour: string = d.getHours() > 9 ? d.getHours().toString() : '0' + d.getHours().toString();
    let min: string = (Math.floor(d.getMinutes() / 5)) > 9 ? (Math.floor(d.getMinutes() / 5)).toString() : '0' + (Math.floor(d.getMinutes() / 5)).toString();
    let reqCode: string = Md5.hashStr(uuid + year + month + day).toString().substr(0, 8);
    let respDay: string = Md5.hashStr(reqCode).toString().substr(0, 8);
    let respHour: string = Md5.hashStr(reqCode + hour).toString().substr(0, 8);
    let resp5min: string = Md5.hashStr(reqCode + hour + min).toString().substr(0, 8);
    let respCode = '';
    return new Promise( async (resolve, reject) => {
      let prompt = await this.alertCtrl.create({
        header: 'Controllo Utente',
        message: 'Codice richiesta: "' + reqCode + '"',
        inputs: [
          {
            name: 'respCode',
            placeholder: 'Codice Risposta'
          }
        ],
        buttons: [
          {
            text: 'Annulla',
            role: 'cancel',
            handler: () => {
              resolve("cancel");
            }
          },
          {
            text: 'Verifica',
            handler: data => {
              this.global.log('ToolsProvider.passwordChecker() - Verifica Codice');
              respCode = data.respCode;
              let result: string = (respCode == respDay || respCode == respHour || respCode == resp5min) ? "ok" : "ko";
              resolve(result);
            }
          }
        ]
      });
      prompt.present();
    });
  }

  parsePrezzo(prezzo: number): string {
    this.global.log('ToolsProvide.parsePrezzo() - started');
    this.global.log('ToolsProvide.parsePrezzo() - prezzo: ' + prezzo);
    let parseprezzo: string = prezzo.toString().replace('.', ',');
    let commai = parseprezzo.indexOf(',');
    if(commai > -1) {
      if(parseprezzo.substr(commai + 1).length < 2) {
        parseprezzo += '0';
      }
    }
    else {
      parseprezzo += ',00';
    }
    return '€ ' + parseprezzo;
  }

  parseError(err: any, level: number, message: string): any {
    let myerr = {level: 0, message: ''};
    if(!err) {
      return {level: level, message: message};
    }
    if(!('level' in err)) {
      myerr.level = level;
    }
    else {
      myerr.level = err.level;
    }
    if(!('message' in err)) {
      if(typeof(err) == 'string') {
        myerr.message = err;
      }
      else {
        let body: string = '';
        if('_body' in err) {
          body = err._body;
        }
        myerr.message = body.length > 0 ? body : message;
      }
    }
    else {
      myerr.message = err.message;
    }
    return myerr;
  }

  nuovoOrdine() {
    this.global.ordineModificato = false;
    this.global.ordine = new Ordine(new TestataOrdine(null, null, 0, this.getDateTimeNow(), '', 0, 0, 0, ''), []);
  }

  ordineInCorso() {
    let c1: boolean = this.global.ordine.righe.length > 0 ;
    let c2: boolean = this.global.ordine.testata == null ? false : true;
    let c3: boolean = c2 ? this.global.ordine.testata.cliente == null ? false : true :  false;
    let c4: boolean = this.global.ordineModificato;
    return c4 && (c1 || c2 || c3);
  }

  getNumRiga(righe: RigaOrdine[]): number {
    let maxn: number = 0;
    for(let r of righe) {
      if(r.numRiga > maxn) maxn = r.numRiga;
    }
    return (maxn + 1);
  }

  insertRiga(articolo: Articolo, prezzo: number, quantita: number, note: string, listino: string): Promise<any> {
    this.global.ordineModificato = true;
    let numRiga: number = this.getNumRiga(this.global.ordine.righe);
    let ro: RigaOrdine = new RigaOrdine(articolo,
                                        this.global.ordine.testata.numOrdine,
                                        numRiga,
                                        this.global.ordine.testata.cliente.codice,
                                        prezzo,
                                        quantita,
                                        listino,
                                        note,
                                        0,
                                        this.getDateTimeNow(),
                                        0,
                                        '',
                                        this.global.getSettings().codiceAgente);
    let rowpresent: boolean = false;
    for(let i = 0; i < this.global.ordine.righe.length; i++) {
      if(this.global.ordine.righe[i].articolo.Codice == ro.articolo.Codice) {
        rowpresent = true;
        this.global.ordine.righe[i].prezzo = ro.prezzo;
        this.global.ordine.righe[i].quantita = ro.quantita;
        this.global.ordine.righe[i].note = ro.note;
      }
    }
    if(!rowpresent) {
      this.global.ordine.righe.push(ro);
    }
    return this.aggiornaTotale();
  }

  removeRiga(riga: RigaOrdine): Promise<any> {
    this.global.ordineModificato = true;
    let rowi = this.global.ordine.righe.indexOf(riga);
    this.global.ordine.righe.splice(rowi, 1);
    for(let i = 0; i < this.global.ordine.righe.length; i++) {
      this.global.ordine.righe[i].numRiga = i + 1;
    }
    return this.aggiornaTotale();
  }

  getTotQuantita(): Promise<number> {
    let totQuantita: number = 0;
    for(let ro of this.global.ordine.righe) {
      totQuantita += ro.quantita;
    }
    return Promise.resolve(totQuantita);
  }

  aggiornaTotale(): Promise<any> {
    let tot: number = 0;
    for(let ro of this.global.ordine.righe) {
      tot = Math.round((tot + this.prezzoROarticolo(ro.prezzo, ro.quantita, ro.articolo.PesoPz, ro.articolo.PzConf)) * 100) / 100;
    }
    this.global.ordine.testata.totale = tot;
    let cliente = this.global.ordine.testata.cliente;
    let totsco = Math.round((tot - (tot * (cliente.sconto1 / 100))) * 100) / 100;
    totsco = Math.round((totsco - (totsco * (cliente.sconto2 / 100))) * 100) / 100;
    totsco = Math.round((totsco - (totsco * (cliente.sconto3 / 100))) * 100) / 100;
    this.global.ordine.testata.totaleScontato = totsco;
    return Promise.resolve();
  }

  prezzoROarticolo(prezzo: number, quantita: number, peso: number, pzconf: number): number {
    peso = peso == 0 ? 1 : peso;
    pzconf = pzconf == 0 ? 1 : pzconf;
    return Math.round((prezzo * quantita * pzconf * peso) * 100) / 100;
  }

  getDateTimeNow() {
    let d = new Date();
    let year = '' + d.getFullYear();
    let month = (d.getMonth() + 1) > 9 ? '' + (d.getMonth() + 1) : '0' + (d.getMonth() + 1);
    let day = d.getDate() > 9 ? '' + d.getDate() : '0' + d.getDate();
    let hh = d.getHours() > 9 ? '' + d.getHours() : '0' + d.getHours();
    let mm = d.getMinutes() > 9 ? '' + d.getMinutes() : '0' + d.getMinutes();
    let ss = d.getSeconds() > 9 ? '' + d.getSeconds() : '0' + d.getSeconds();
    let datestr = year + '-' + month + '-' + day + ' ' + hh + ':' + mm + ':' + ss;
    this.global.log('ToolsProvider.getDateTimeNow() - Date: ' + datestr);
    return datestr;
  }

  getDateNow() {
    let d = new Date();

    let month = (d.getMonth() + 1) > 9 ? '' + (d.getMonth() + 1) : '0' + (d.getMonth() + 1);
    let day = d.getDate() > 9 ? '' + d.getDate() : '0' + d.getDate();
    let datestr = year + '-' + month + '-' + day;
    this.global.log('ToolsProvider.getDateTimeNow() - Date: ' + datestr);
    return datestr;
  }

  dateSQLtoLocal(date: string, sep: string = '/'): string {
    let dd = date.substr(8,2);
    let mm = date.substr(5,2);
    let yyyy = date.substr(0,4);
    return dd + sep + mm + sep + yyyy;
  }

  parseJSON(res: Response): any {
    if(res.text().indexOf('MySQL->Connect') > -1) {
      throw new Error('Errore di connessione al DB.');
    }
    if(res.text().indexOf('<') > -1 || res.text().indexOf('>') > -1) {
      throw new Error('Errore interno al server.');
    }
    return res.json();
  }

  urlEndSlash(url: string): string {
    if(url.charAt(url.length - 1) !== '/') return url + '/';
    else return url;
  }

  urlHttpProtocol(url: string): string {
    if(url.indexOf('proxy') > -1) return url;
    if(url.startsWith('http://')) return url;
    if(url.startsWith('https://')) return url;
    else return 'https://' + url;
  }

  crudTransform(tables: any): any {
    this.global.log('ToolProvider.crudTransform() - started');
    let array_flip = function(trans) {
      let key, tmp_ar = {};
      for(key in trans) {
        tmp_ar[trans[key]] = key;
      }
      return tmp_ar;
    };
    let get_objects = function(tables: any, table_name: any, where_index?: any, match_value?: any): any {
      let objects = [];
      for(let record in tables[table_name]['records']) {
        record = tables[table_name]['records'][record];
        if(!where_index || record[where_index]==match_value) {
          let object = {};
          for(let index in tables[table_name]['columns']) {
            let column = tables[table_name]['columns'][index];
            object[column] = record[index];
            for(let relation in tables) {
              let reltable = tables[relation];
              for(let key in reltable['relations']) {
                let target = reltable['relations'][key];
                if(target == table_name+'.'+column) {
                  let column_indeces = array_flip(reltable['columns']);
                  object[relation] = get_objects(tables, relation, column_indeces[key], record[index]);
                }
              }
            }
          }
          objects.push(object);
        }
      }
      return objects;
    };
    let tree = {};
    for(let name in tables) {
      let table = tables[name];
      if(!table['relations']) {
        tree[name] = get_objects(tables, name);
        if(table['results']) {
          tree['_results'] = table['results'];
        }
      }
    }
    return tree;
  }
